def manageHits(receivedAttack)
    lose = true
    defense = defensiveEnergy
    if defense < receivedAttack
        gotWounded
        incConsecutiveHits
    else
        resetHits
    end
    if consecutiveHits == HITS2LOSE || dead 
        resetHits
        lose = true
    else
        lose = false
    end
    return lose
end

def move(direction, validMoves) 
    size = validMoves.size
    contained = false
    firstElement = Directions
    
    validMoves.each do |d|
        if d==direction 
            contained=true
        end
        if size > 0 && !contained
            firstElement = validMoves[0]
        else    
            firstElement = direction
        end
        
    end
    firstElement
end

def receiveReward
    wReward = Dice.weaponReward
    sReward = Dice.shieldReward

    for i in 0...wReward
        wnew = newWeapon
        receiveWeapon(wnew)
    end
    for i in 0...sReward
        snew = newShield
        receiveShield(snew)
    end
    extraHealth = Dice.healthReward
    @health = @health + extraHealth
end

def to_s
    output = ""
    output = "Player[name: #{@name}, intelligence: #{@intelligence}, strength: #{@strength}, health: #{@health}, row: #{@row}, col: #{@col}]"
    output += "\nWeapons:"
    weapons.each do |w|
        output += "\n#{w}"
    end
    output += "\nShields:"
    shields.each do |s|
        output += "\n#{s}"
    end
    output
end
