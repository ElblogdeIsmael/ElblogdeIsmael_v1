Overview

Representa el estado actual del juego, incluyendo el laberinto, jugadores, monstruos, el jugador actual, el ganador, y el registro de eventos (log).

def initialize(labyrinth, players, monsters, currentPlayer, winner, log)
    @labyrinth = labyrinth
    @players = players
    @monsters = monsters
    @currentPlayer = currentPlayer
    @winner = winner
    @log = log
end

def to_s
    "\nlabyrinth=#{labyrinth}" +
    "\n \nplayers=#{players}" +
    "\n \nmonsters=#{monsters}" +
    "\n \ncurrent_player=#{currentPlayer}" +
    "\n \nwinner=#{winner}" +
    "\n \nlog=#{log}" +
    "\n"
end
